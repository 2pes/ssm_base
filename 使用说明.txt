## 1. 数据库
```sql
-- 添加新用户
create user 'smmtest'@'%' identified by 'ssmtest';
-- 为用户创建数据库
create database ssm_base DEFAULT CHARSET utf8 COLLATE utf8_general_ci;
-- 授予用户通过外网IP对于该数据库的全部权限
grant all privileges on `ssm_base`.* to 'smmtest'@'%' identified by 'smmtest';
-- 刷新权限
flush privileges;
```
